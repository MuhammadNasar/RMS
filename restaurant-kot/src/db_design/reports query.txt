SELECT `id` AS `bill_number`, `total_amount_receivable`, `discount_amount`, `net_amount`, `payment_method`, DATE_FORMAT(`bill_date_time`, '%d %b %Y') AS `bill_date` FROM `restaurant_kot` WHERE DATE_FORMAT(`bill_date_time`, '%Y-%m-%d') BETWEEN '2018-07-25' AND '2018-08-03' AND `payment_method` ='CASH'
